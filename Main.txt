-- // Servi√ßos
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local player = Players.LocalPlayer
local backpack = player:WaitForChild("Backpack")
local itemIDs = {
    17770317484,
    17772174303,
    17822722698,
    17771175724,
    17812415139,
    17812417356,
    17822749561,
    892267099,
    892265784,
    892268340,
    11274394350
}

-- // Fun√ß√£o para equipar itens e disparar remote
local function equipItems()
    -- Remove duplicatas
    local seen = {}
    local uniqueList = {}
    for _, id in ipairs(itemIDs) do
        if not seen[id] then
            seen[id] = true
            table.insert(uniqueList, id)
        end
    end

    -- Equipar cada item
    for _, id in ipairs(uniqueList) do
        local success, err = pcall(function()
            ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("Wear"):InvokeServer(id)
        end)
        if success then
            print("‚úÖ Item ID "..id.." equipado com sucesso!")
        else
            warn("‚ö†Ô∏è Falha ao equipar Item ID "..id.." | "..tostring(err))
        end
        task.wait(0.3)
    end

    -- Executa o ChangeBodyColor
    local bodyColorRemote = ReplicatedStorage:WaitForChild("Remotes"):FindFirstChild("ChangeBodyColor")
    if bodyColorRemote then
        local args = { [1] = "Really black" }
        bodyColorRemote:FireServer(unpack(args))
        print("üé® Remote ChangeBodyColor executado com sucesso!")
    else
        warn("‚ùå Remote ChangeBodyColor n√£o encontrado!")
    end
end

-- // Cria a Tool
local tool = Instance.new("Tool")
tool.Name = "apare√ßer"
tool.RequiresHandle = false
tool.CanBeDropped = false
tool.Parent = backpack

-- // Conectar evento de clique
tool.Activated:Connect(function()
    equipItems()
end)

print("üõ†Ô∏è Tool criada no Backpack! Clique para equipar os itens.")
